## –ö–ª–∞—Å—Å `Clock`

```python
class Clock:
    __DAY = 86400  # —á–∏—Å–ª–æ —Å–µ–∫—É–Ω–¥ –≤ 1 –¥–Ω–µ

    def __init__(self, seconds: int):
        if not isinstance(seconds, int):
            raise TypeError("–°–µ–∫—É–Ω–¥—ã –¥–æ–ª–∂–Ω—ã –±—ã—Ç—å —Ü–µ–ª—ã–º —á–∏—Å–ª–æ–º")
        self.seconds = seconds % self.__DAY  # –æ—Å—Ç–∞—Ç–æ–∫ –æ—Ç –¥–µ–ª–µ–Ω–∏—è –∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç, —á—Ç–æ `seconds` –≤—Å–µ–≥–¥–∞ –Ω–∞—Ö–æ–¥–∏—Ç—Å—è –≤ –ø—Ä–µ–¥–µ–ª–∞—Ö –æ—Ç `0` –¥–æ `86399` (—Å–µ–∫—É–Ω–¥—ã –≤ —Å—É—Ç–∫–∞—Ö)

    def __eq__(self, other):
	    # –≠—Ç–æ—Ç –º–µ—Ç–æ–¥ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –ø–æ–≤–µ–¥–µ–Ω–∏–µ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `==`
        if not isinstance(other, (int, Clock)):
	        # –ï—Å–ª–∏ `other` ‚Äì –Ω–µ `int` –∏ –Ω–µ `Clock`, –≤—ã–±—Ä–∞—Å—ã–≤–∞–µ—Ç—Å—è `TypeError`
            raise TypeError('–û–ø–µ—Ä–∞–Ω–¥ —Å–ø—Ä–∞–≤–∞ –¥–æ–ª–∂–µ–Ω –∏–º–µ—Ç—å —Ç–∏–ø int –∏–ª–∏ Clock')
		# - –ï—Å–ª–∏ `other` ‚Äì —á–∏—Å–ª–æ (`int`), —Ç–æ —Å—Ä–∞–≤–Ω–∏–≤–∞–µ–º —Å `self.seconds`
		# - –ï—Å–ª–∏ `other` ‚Äì –æ–±—ä–µ–∫—Ç `Clock`, —Ç–æ —Å—Ä–∞–≤–Ω–∏–≤–∞–µ–º `self.seconds` —Å `other.seconds`
        sc = other if isinstance(other, int) else other.seconds
        return self.seconds == sc

    def __lt__(self, other):
        """–≠—Ç–æ—Ç –º–µ—Ç–æ–¥ –æ–ø—Ä–µ–¥–µ–ª—è–µ—Ç –ø–æ–≤–µ–¥–µ–Ω–∏–µ –æ–ø–µ—Ä–∞—Ç–æ—Ä–∞ `<`
        –ü–æ —Ö–æ—Ä–æ—à–µ–º—É –Ω—É–∂–Ω–æ –≤—ã–≤–µ—Å—Ç–∏ –≤ –æ—Ç–¥–µ–ª—å–Ω—ã–π –º–µ—Ç–æ–¥, —á—Ç–æ–±—ã –∏–∑–±–µ–∂–∞—Ç—å –¥—É–±–ª–∏—Ä–æ–≤–∞–Ω–∏—è"""
        if not isinstance(other, (int, Clock)): # –ï—Å–ª–∏ –Ω–µ–ø–æ–¥—Ö–æ–¥—è—â–∏–π —Ç–∏–ø, –≤—ã–∑—ã–≤–∞–µ—Ç—Å—è –æ—à–∏–±–∫–∞
            raise TypeError('–û–ø–µ—Ä–∞–Ω–¥ —Å–ø—Ä–∞–≤–∞ –¥–æ–ª–∂–µ–Ω –∏–º–µ—Ç—å —Ç–∏–ø int –∏–ª–∏ Clock')
		# –ê–Ω–∞–ª–æ–≥–∏—á–Ω–æ–µ —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ –∫–∞–∫ –∏ –≤ __eq__
        sc = other if isinstance(other, int) else other.seconds
        return self.seconds < sc
```

### –ü—Ä–∏–º–µ—Ä—ã –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è

```python
c1 = Clock(1000)
c2 = Clock(1000)
c3 = Clock(2000)

print(c1 == c2)  # True (–µ—Å–ª–∏ –±—ã –≤ –∫–ª–∞—Å—Å–µ –Ω–µ –±—ã–ª–æ __eq__, —Ç–æ –±—É–¥–µ—Ç False, —Ç.–∫. —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –ø–æ id –æ–±—ä–µ–∫—Ç–æ–≤)
print(c1 == 1000)  # True (—Ç–∞–∫ —Ç–æ–∂–µ –º–æ–∂–Ω–æ)

# –û–ø–µ—Ä–∞—Ç–æ—Ä != —Ä–∞–±–æ—Ç–∞–µ—Ç –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ –¥–∞–∂–µ –±–µ–∑ —è–≤–Ω–æ–≥–æ –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è __ne__()
print(c1 != c2)  # False

print(c1 < c3)  # True

# –ï—Å–ª–∏ –æ–ø–µ—Ä–∞—Ç–æ—Ä > –Ω–µ —Ä–µ–∞–ª–∏–∑–æ–≤–∞–Ω, Python –∑–∞–º–µ–Ω–∏—Ç –µ–≥–æ –Ω–∞ c2 < c1, —á—Ç–æ –∫–æ—Ä—Ä–µ–∫—Ç–Ω–æ —Ä–∞–±–æ—Ç–∞–µ—Ç
print(c3 > c1)  # True
```

### **–í—ã–≤–æ–¥—ã**

1. **–ú–∞–≥–∏—á–µ—Å–∫–∏–µ –º–µ—Ç–æ–¥—ã `__eq__` –∏ `__lt__` –ø–æ–∑–≤–æ–ª—è—é—Ç –æ–±—ä–µ–∫—Ç–∞–º —É—á–∞—Å—Ç–≤–æ–≤–∞—Ç—å –≤ —Å—Ä–∞–≤–Ω–µ–Ω–∏—è—Ö.**
2. **–ú–µ—Ç–æ–¥ `__ne__` (`!=`) —Ä–∞–±–æ—Ç–∞–µ—Ç –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ —á–µ—Ä–µ–∑ `not __eq__`.**
3. **–ú–µ—Ç–æ–¥—ã `>`, `>=`, `<=` —Ç–æ–∂–µ —Ä–∞–±–æ—Ç–∞—é—Ç, –µ—Å–ª–∏ –µ—Å—Ç—å `__lt__` –∏–ª–∏ `__eq__`, –±–ª–∞–≥–æ–¥–∞—Ä—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–º—É –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏—é.**
4. **–õ—É—á—à–µ –≤—ã–Ω–µ—Å—Ç–∏ –ø—Ä–æ–≤–µ—Ä–∫—É —Ç–∏–ø–∞ (`isinstance`) –≤ –æ—Ç–¥–µ–ª—å–Ω—ã–π –º–µ—Ç–æ–¥, —á—Ç–æ–±—ã –Ω–µ –¥—É–±–ª–∏—Ä–æ–≤–∞—Ç—å –∫–æ–¥.**
